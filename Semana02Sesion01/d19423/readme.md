
### Comandos Basicos

``` bash
#cambiar directorio
cd rutaAbsoluta
cd rutaRelativa (../)

#donde estoy
pwd

#crear carpeta
mkdir nombreCarpeta

touch "mi primer archivo"

rm "mi primer archivo"

#documentcion del comando

man comando

#listar directorio 
ls

ls -la

#pa copiar
cp origen destino

#para mover
mv origen destino

#para leer 
cat nombrearchivo

#para escribir 

echo "texto" > nombrearchivo

```
